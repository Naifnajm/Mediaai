Client الأساسي

// lib/odoo/client.ts

class OdooClient {
  private baseUrl: string;
  private db: string = 'MediaAi';

  constructor(baseUrl: string) {
    this.baseUrl = baseUrl;
  }

  // ── تسجيل الدخول ──────────────────────────────────
  async authenticate(username: string, password: string) {
    const res = await this.jsonRpc('/web/session/authenticate', {
      db: this.db, login: username, password
    });
    return res.result?.uid ? true : false;
  }

  // ── قراءة قائمة سجلات ─────────────────────────────
  async searchRead<T>(
    model: string,
    domain: any[] = [],
    fields: string[] = [],
    options: { limit?: number; offset?: number; order?: string } = {}
  ): Promise<T[]> {
    return this.call({
      model, method: 'search_read',
      args: [domain],
      kwargs: { fields, limit: options.limit ?? 80,
                offset: options.offset ?? 0,
                order: options.order ?? 'id desc' }
    });
  }

  // ── قراءة سجل واحد ────────────────────────────────
  async read<T>(model: string, id: number, fields: string[]) {
    const results = await this.call<T[]>({
      model, method: 'read', args: [[id], fields]
    });
    return results[0];
  }

  // ── إنشاء سجل جديد ────────────────────────────────
  async create(model: string, values: Record<string, any>): Promise<number> {
    return this.call({ model, method: 'create', args: [values] });
  }

  // ── تعديل سجل ─────────────────────────────────────
  async write(model: string, ids: number[], values: Record<string, any>) {
    return this.call({ model, method: 'write', args: [ids, values] });
  }

  // ── حذف سجل ───────────────────────────────────────
  async unlink(model: string, ids: number[]) {
    return this.call({ model, method: 'unlink', args: [ids] });
  }

  // ── عدد السجلات ───────────────────────────────────
  async searchCount(model: string, domain: any[] = []) {
    return this.call({ model, method: 'search_count', args: [domain] });
  }

  // ── تنفيذ زر / أكشن ───────────────────────────────
  async callButton(model: string, method: string, ids: number[]) {
    return this.call({ model, method, args: [ids] });
  }

  // ── رفع ملف ───────────────────────────────────────
  async uploadAttachment(file: File, resModel: string, resId: number) {
    const formData = new FormData();
    formData.append('ufile', file);
    formData.append('model', resModel);
    formData.append('id', resId.toString());
    const res = await fetch(`${this.baseUrl}/web/binary/upload_attachment`, {
      method: 'POST', body: formData, credentials: 'include'
    });
    return res.json();
  }

  // ── جلب المرفقات ──────────────────────────────────
  async getAttachments(resModel: string, resId: number) {
    return this.searchRead('ir.attachment', [
      ['res_model', '=', resModel], ['res_id', '=', resId]
    ], ['id', 'name', 'mimetype', 'file_size', 'create_date']);
  }

  // ── تنفيذ JSON-RPC ────────────────────────────────
  private async call<T>(options: any): Promise<T> {
    const response = await this.jsonRpc('/web/dataset/call_kw', {
      model: options.model, method: options.method,
      args: options.args ?? [],
      kwargs: {
        context: { lang: 'ar_001', tz: 'Asia/Riyadh' },
        ...options.kwargs
      }
    });
    if (response.error) throw new Error(response.error.data?.message);
    return response.result;
  }

  private async jsonRpc(endpoint: string, params: any) {
    const res = await fetch(`${this.baseUrl}${endpoint}`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      credentials: 'include',
      body: JSON.stringify({ jsonrpc: '2.0', method: 'call',
                             id: Date.now(), params })
    });
    return res.json();
  }
}

export const odoo = new OdooClient(process.env.NEXT_PUBLIC_ODOO_URL!);

Custom Hooks

// hooks/useOdooList.ts
export function useOdooList<T>(model, domain = [], fields = [], opts = {}) {
  return useQuery({
    queryKey: ['odoo', model, domain, fields],
    queryFn: () => odoo.searchRead<T>(model, domain, fields, opts),
    staleTime: 30_000,
  });
}

// hooks/useOdooRecord.ts
export function useOdooRecord<T>(model, id, fields) {
  return useQuery({
    queryKey: ['odoo', model, id],
    queryFn: () => odoo.read<T>(model, id, fields),
    enabled: !!id,
  });
}

// hooks/useOdooCreate.ts
export function useOdooCreate(model) {
  const qc = useQueryClient();
  return useMutation({
    mutationFn: (values) => odoo.create(model, values),
    onSuccess: () => qc.invalidateQueries({ queryKey: ['odoo', model] }),
  });
}

// lib/odoo/bus.ts
class OdooBus {
  private ws: WebSocket | null = null;
  private listeners = new Map<string, Function[]>();

  connect() {
    this.ws = new WebSocket(`wss://${ODOO_HOST}/websocket`);
    this.ws.onmessage = (event) => {
      const messages = JSON.parse(event.data);
      messages.forEach((msg: any) => {
        const handlers = this.listeners.get(msg.type) ?? [];
        handlers.forEach(fn => fn(msg.payload));
      });
    };
  }

  subscribe(channel: string, callback: Function) {
    if (!this.listeners.has(channel)) this.listeners.set(channel, []);
    this.listeners.get(channel)!.push(callback);
  }
}
